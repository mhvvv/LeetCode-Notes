[1765. 地图中的最高点](https://leetcode-cn.com/problems/map-of-highest-peak/) 2022-01-29 每日一题

bfs，从水域出发，一层一层向外拓展即可.

```C++
class Solution {
public:
    int next[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    vector<vector<int>> highestPeak(vector<vector<int>>& isWater) {
        int m = isWater.size(), n = isWater[0].size(); 
        vector<vector<bool>> vis(m, vector<bool>(n, false));
        queue<pair<int,int>> q;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(isWater[i][j] == 1) {
                    isWater[i][j] = 0;
                    q.push({i,j});
                    vis[i][j] = true;
                }
            }
        }
        int x, y, layer = 0;
        while(!q.empty()) {
            int SIZE = q.size();
            ++layer;
            for(int i = 0; i < SIZE; ++i) {
                pair<int, int> p = q.front();
                q.pop();
                for(int j = 0; j < 4; ++j) {
                    x = p.first + next[j][0];
                    y = p.second + next[j][1];
                    if(x < m && x >= 0 && y < n && y >= 0 && !vis[x][y]) {
                        isWater[x][y] = layer;
                        q.push({x, y});
                        vis[x][y] = true;
                    }
                }          
            }
        }
        return isWater;
    }
};

O(mn), O(mn)
```
