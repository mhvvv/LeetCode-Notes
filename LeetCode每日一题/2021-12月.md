class Solution {
public:
    int findJudge(int n, vector<vector<int>>& trust) {
        vector<int> u(n+1, 0), v(n+1, 0);
        for(vector<int> i : trust) {
            ++u[i.front()];
            ++v[i.back()];
        }
        for(int i=1; i<=n; ++i) {
            if(u[i] == 0 && v[i] == n-1)
                return i;
        }
        return -1;
    }
};
